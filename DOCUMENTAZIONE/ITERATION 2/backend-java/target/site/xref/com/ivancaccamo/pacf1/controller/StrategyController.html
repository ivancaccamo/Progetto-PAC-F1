4   import com.ivancaccamo.pacf1.model.RaceStrategy;

6   import com.ivancaccamo.pacf1.repository.StrategyRepository;
7   import com.ivancaccamo.pacf1.service.OptimizationEngine;

9   import org.springframework.beans.factory.annotation.Autowired;
10  import org.springframework.web.bind.annotation.*;
11  
12  import java.util.ArrayList;
13  import java.util.List;
14  
15  /**
16   * Controller REST principale dell'applicazione SPS-F1.
17   * <p>
18   * Questa classe agisce come punto di ingresso per le API (Facade Pattern),
19   * orchestrando la comunicazione tra il frontend, il servizio di Machine Learning (Python)
20   * e il motore di ottimizzazione interno (Java).
21   * Gestisce inoltre le operazioni CRUD per il salvataggio dello storico delle strategie.
22   * </p>
23   *
24   * @author Team SPS-F1
25   */
26  @RestController
27  @RequestMapping("/api")
28  public class StrategyController {
29  
30      @Autowired
31      private PythonMLService mlService;
32  
33      @Autowired
34      private OptimizationEngine optimizer;
35  
36      @Autowired
37      private StrategyRepository repository;
38  
39      /**
40       * Calcola le migliori strategie di gara basandosi sui parametri ambientali forniti.
41       * <p>
42       * Il flusso di esecuzione Ã¨ il seguente:
43       * 1. Invia i dati meteo al microservizio Python per ottenere le curve di degrado.
44       * 2. Se le predizioni sono valide, invoca l'OptimizationEngine.
45       * 3. Restituisce le Top 3 strategie ottimali calcolate tramite Programmazione Dinamica.
46       * </p>
47       *
48       * @param circuit   Il nome del circuito (es. "Bahrain Grand Prix").
49       * @param laps      Il numero totale di giri della gara.
50       * @param airTemp   La temperatura dell'aria in gradi Celsius.
51       * @param trackTemp La temperatura dell'asfalto in gradi Celsius.
52       * @return Una lista di oggetti {@link RaceStrategy} contenente le strategie suggerite,
53       * o una lista vuota se il servizio ML non risponde.
54       */
55      // 1. Calcola Strategia
56      @GetMapping("/strategy")
57      public List<RaceStrategy> getStrategy(
58              @RequestParam(defaultValue = "Bahrain Grand Prix") String circuit,
59              @RequestParam(defaultValue = "57") int laps,
60              @RequestParam(defaultValue = "30.0") double airTemp,
61              @RequestParam(defaultValue = "45.0") double trackTemp) {
62  
63          PredictionResponse predictions = mlService.getPrediction(circuit, airTemp, trackTemp);
64          if (predictions == null) return new ArrayList<>();
65          return optimizer.calculateTop3Strategies(laps, predictions.getPredictions());
66      }
67  
68      /**
69       * Salva una specifica strategia nel database persistente.
70       * <p>
71       * Permette all'utente di archiviare una simulazione ritenuta interessante
72       * per poterla consultare successivamente nella sezione "Archivio".
73       * </p>
74       *
75       * @param strategy L'oggetto {@link SavedStrategy} contenente i dettagli della simulazione da salvare.
76       * @return L'oggetto salvato, comprensivo dell'ID generato dal database.
77       */
78      // 2. Salva una strategia nel DB
79      @PostMapping("/history")
80      public SavedStrategy saveStrategy(@RequestBody SavedStrategy strategy) {
81          return repository.save(strategy);
82      }
83  
84      /**
85       * Recupera l'intero storico delle simulazioni salvate.
86       *
87       * @return Una lista di tutte le strategie presenti nel database.
88       */
89      // 3. Leggi tutto lo storico
90      @GetMapping("/history")
91      public List<SavedStrategy> getHistory() {
92          return repository.findAll();
93      }
94  
95      /**
96       * Elimina una strategia dall'archivio.
97       *
98       * @param id L'identificativo univoco (Primary Key) della strategia da rimuovere.
99       */
100     // 4. Cancella una strategia dall'archivio
101     @DeleteMapping("/history/{id}")
102     public void deleteHistory(@PathVariable Long id) {
103         repository.deleteById(id);
104     }
105 }
