10  /**
11   * Motore algoritmico principale del sistema SPS-F1.
12   * <p>
13   * Questa classe implementa la logica di ottimizzazione per determinare la strategia di gara ideale.
14   * Utilizza un approccio di <b>Programmazione Dinamica (DP)</b> con Memoization per esplorare
15   * l'albero delle possibili decisioni (mescola, lunghezza stint) e trovare il cammino minimo
16   * in termini di tempo totale.
17   * </p>
18   * <p>
19   * L'algoritmo tiene conto del degrado degli pneumatici, del tempo perso in pit-lane e
20   * dei vincoli regolamentari (obbligo di usare almeno due mescole diverse).
21   * </p>
22   *
23   * @author Team SPS-F1
24   */
25  @Service
26  public class OptimizationEngine {
27  
28      /**
29       * Tempo medio perso per effettuare un pit-stop (percorrenza pit-lane + cambio gomme).
30       * Valore fisso stimato a 20.0 secondi.
31       */
32      private static final double PIT_STOP_LOSS = 20.0;
33  
34      /**
35       * Valore sentinella per indicare un tempo o un costo infinito (strategia non valida).
36       */
37      private static final double INFINITY = 1e9; // Un numero grandissimo
38  
39      /**
40       * Cache per la Memoization.
41       * Mappa uno stato univoco (giro corrente + maschera gomme usate) al miglior tempo ottenibile da quello stato in poi.
42       * Chiave: "giroCorrente-mascheraGommeUsate" -> Valore: Miglior Tempo
43       */
44      // Cache per la Memoization: salva i risultati parziali per non ricalcolarli
45      // Chiave: "giroCorrente-mascheraGommeUsate" -> Valore: Miglior Tempo
46      private Map<String, Double> memo = new HashMap<>();
47      
48      /**
49       * Mappa utilizzata per tracciare le decisioni ottimali prese ad ogni passo.
50       * Fondamentale per la fase di backtracking che ricostruisce la lista degli stint finali.
51       */
52      // Per ricostruire la strategia alla fine
53      private Map<String, StintDecision> bestDecisions = new HashMap<>();
54  
55      /**
56       * Metodo principale per il calcolo delle strategie.
57       * <p>
58       * Esegue una ricerca esaustiva intelligente per trovare le Top 3 strategie migliori.
59       * Invece di lanciare una singola esecuzione DP, itera su tutte le possibili combinazioni
60       * di "Primo Stint" (mescola iniziale e durata) e delega alla DP l'ottimizzazione del resto della gara.
61       * Questo approccio ibrido permette di diversificare i risultati e trovare alternative valide
62       * (es. strategia a 1 sosta vs 2 soste).
63       * </p>
64       *
65       * @param totalLaps Il numero totale di giri della gara.
66       * @param tyres     La lista delle predizioni di degrado per le mescole disponibili.
67       * @return Una lista contenente le migliori 3 strategie uniche, ordinate per tempo totale crescente.
68       */
69      public List<RaceStrategy> calculateTop3Strategies(int totalLaps, List<TyrePrediction> tyres) {
70          System.out.println("--- AVVIO ALGORITMO DP (N SOSTE) - RICERCA ESAUSTIVA TOP 3 ---");
71          
72          List<RaceStrategy> candidates = new ArrayList<>();
73          if (tyres.isEmpty()) return candidates;
74  
75          // Invece di chiedere subito "qual è il meglio assoluto",
76          // proviamo manualmente ogni possibile PRIMO STINT (Mescola + Durata)
77          // e chiediamo alla DP di risolvere ottimamente il RESTO della gara.
78          
79          for (int i = 0; i < tyres.size(); i++) {
80              TyrePrediction startTyre = tyres.get(i);
81              int startMask = (1 << i); // Maschera con la prima gomma usata
82  
83              // Proviamo diverse durate per il primo stint (es. da 1 giro fino a metà gara)
84              // Usiamo uno step di 3 giri per avere varietà senza troppi calcoli
85              int minStint = Math.max(2, (int)(totalLaps * 0.15));
86              int maxStint = Math.max(minStint + 2, totalLaps - 2);
87              int step = Math.max(1, totalLaps / 15); // Step adattivo
88  
89              for (int firstStintLaps = 1; firstStintLaps < totalLaps - 1; firstStintLaps++) {
90                  
91                  // IMPORTANTE: Puliamo la memoria per ogni tentativo, altrimenti riusa calcoli vecchi
92                  memo.clear();
93                  bestDecisions.clear();
94  
95                  // Calcoliamo il costo del primo stint manuale
96                  double firstStintTime = calculateStintTime(startTyre, firstStintLaps);
97  
98                  // Chiediamo alla DP: "Qual è il tempo minimo per finire la gara da qui in poi?"
99                  // Nota: solve() aggiungerà automaticamente il costo del pit stop iniziale
100                 double timeRest = solve(firstStintLaps, startMask, totalLaps, tyres);
101 
102                 if (timeRest < INFINITY) {
103                     // Ricostruiamo la strategia per la seconda parte
104                     RaceStrategy restStrategy = reconstructStrategy(firstStintLaps, startMask, totalLaps, tyres);
105                     
106                     if (restStrategy != null) {
107                         // Creiamo la strategia completa unendo Primo Stint + Resto
108                         RaceStrategy fullStrategy = new RaceStrategy();
109                         
110                         // Tempo totale = Primo Stint + Resto (che include i pit stop successivi)
111                         fullStrategy.setTotalTime(firstStintTime + timeRest);
112                         
113                         // Aggiungiamo il primo stint in testa alla lista
114                         fullStrategy.getStints().add(new Stint(startTyre.getCompound(), 1, firstStintLaps));
115                         fullStrategy.getStints().addAll(restStrategy.getStints());
116                         
117                         // Calcoliamo le soste (Numero di stint - 1)
118                         fullStrategy.setPitStops(fullStrategy.getStints().size() - 1);
119 
120                         candidates.add(fullStrategy);
121                     }
122                 }
123             }
124         }
125 
126         // 3. ORDINIAMO E FILTRIAMO
127         // Ordiniamo per tempo totale crescente
128         candidates.sort(Comparator.comparingDouble(RaceStrategy::getTotalTime));
129 
130         // Rimuoviamo duplicati (strategie con lo stesso tempo quasi identico)
131         List<RaceStrategy> uniqueResults = new ArrayList<>();
132         Set<Integer> seenTimes = new HashSet<>(); // Usiamo int per arrotondare e filtrare
133 
134         for (RaceStrategy s : candidates) {
135             int timeInt = (int) s.getTotalTime();
136             // Se non abbiamo già una strategia con questo tempo esatto, la aggiungiamo
137             if (!seenTimes.contains(timeInt)) {
138                 uniqueResults.add(s);
139                 seenTimes.add(timeInt);
140             }
141             if (uniqueResults.size() >= 3) break; // Ci fermiamo a 3
142         }
143 
144         return uniqueResults;
145     }
146     /**
147      * FUNZIONE RICORSIVA CORE (DP).
148      * <p>
149      * Questo metodo implementa il cuore dell'algoritmo di Programmazione Dinamica.
150      * Calcola il tempo minimo necessario per completare la gara partendo da uno stato specifico
151      * definito dal giro corrente e dalle mescole già utilizzate.
152      * </p>
153      * <p>
154      * Utilizza la tecnica della <b>Memoization</b>: prima di calcolare una soluzione, controlla
155      * se lo stato (currentLap, usedTyresMask) è già stato risolto e salvato nella mappa {@code memo}.
156      * Se è un nuovo stato, esplora tutte le possibili decisioni future (quale gomma montare e per quanti giri),
157      * sceglie quella che minimizza il tempo totale e salva la decisione migliore.
158      * </p>
159      *
160      * @param currentLap    Il giro attuale da cui inizia il prossimo stint (stato temporale).
161      * @param usedTyresMask Una bitmask intera che traccia lo storico delle mescole usate.
162      * (es. bit 0 = Soft, bit 1 = Medium, bit 2 = Hard).
163      * Indispensabile per verificare il regolamento delle due mescole.
164      * @param totalLaps     Il numero totale di giri della gara.
165      * @param tyres         La lista delle predizioni disponibili per le gomme.
166      * @return Il tempo minimo stimato in secondi per arrivare alla fine della gara da questo punto,
167      * oppure {@code INFINITY} se non esiste una strategia valida.
168      */
169     private double solve(int currentLap, int usedTyresMask, int totalLaps, List<TyrePrediction> tyres) {
170         // CASO BASE: Gara finita
171         if (currentLap == totalLaps) {
172             // Controlliamo la regola: abbiamo usato almeno 2 mescole diverse?
173             // Contiamo i bit a 1 nella maschera
174             if (Integer.bitCount(usedTyresMask) >= 2) {
175                 return 0; // Costo 0 per finire (abbiamo già finito)
176             } else {
177                 return INFINITY; // Strategia illegale, penalità infinita
178             }
179         }
180 
181         // MEMOIZATION: Se abbiamo già calcolato questo stato, restituiamo il valore salvato
182         String stateKey = currentLap + "-" + usedTyresMask;
183         if (memo.containsKey(stateKey)) {
184             return memo.get(stateKey);
185         }
186 
187         double minTime = INFINITY;
188         StintDecision bestDecision = null;
189 
190         // PROVIAMO TUTTE LE POSSIBILI MOSSE (Next Stint)
191         // Iteriamo su ogni mescola disponibile
192         for (int i = 0; i < tyres.size(); i++) {
193             TyrePrediction tyre = tyres.get(i);
194             
195             // Maschera aggiornata se usiamo questa gomma (1 << i accende il bit i-esimo)
196             int nextMask = usedTyresMask | (1 << i);
197 
198             // Proviamo tutte le lunghezze possibili per questo stint
199             // Minimo 5 giri, massimo fino alla fine della gara
200             for (int laps = 10; laps <= (totalLaps - currentLap); laps += 1) { // Step 1 per precisione massima
201                 
202                 int nextLap = currentLap + laps;
203                 
204                 // Costo = Tempo guida + Pit Stop (se non è la partenza)
205                 double driveTime = calculateStintTime(tyre, laps);
206                 double pitCost = (currentLap == 0) ? 0 : PIT_STOP_LOSS;
207                 
208                 // RICORSIONE: Tempo totale = costo attuale + costo migliore dal prossimo giro in poi
209                 double timeToFinish = solve(nextLap, nextMask, totalLaps, tyres);
210                 double totalTime = driveTime + pitCost + timeToFinish;
211 
212                 if (totalTime < minTime) {
213                     minTime = totalTime;
214                     // Memorizziamo la decisione presa per ricostruire il percorso dopo
215                     bestDecision = new StintDecision(tyre.getCompound(), laps, nextMask);
216                 }
217             }
218         }
219 
220         // Salviamo il risultato in memoria
221         memo.put(stateKey, minTime);
222         if (bestDecision != null) {
223             bestDecisions.put(stateKey, bestDecision);
224         }
225 
226         return minTime;
227     }

229     /**
230      * Ricostruisce la strategia ottinale completa a partire dai dati salvati.
231      * <p>
232      * Una volta che il metodo {@code solve()} ha popolato la mappa {@code bestDecisions},
233      * questo metodo "naviga" attraverso le decisioni migliori salvate per trasformarle
234      * in una lista ordinata di oggetti {@link Stint} comprensibile per l'utente.
235      * </p>
236      *
237      * @param startLap  Il giro di partenza della ricostruzione.
238      * @param startMask La maschera delle gomme iniziale.
239      * @param totalLaps Il numero totale di giri.
240      * @param tyres     Le informazioni sulle gomme.
241      * @return Un oggetto {@link RaceStrategy} completo, o {@code null} se il percorso è interrotto.
242      */
243     // Ricostruisce la strategia seguendo le "briciole di pane" lasciate dalla DP
244     private RaceStrategy reconstructStrategy(int startLap, int startMask, int totalLaps, List<TyrePrediction> tyres) {
245         RaceStrategy strategy = new RaceStrategy();
246         int currentLap = startLap;
247         int mask = startMask;
248         int stops = -1; 
249         
250         // Se partiamo da metà gara (es. per le alternative), i pit stop vanno contati diversamente
251         if (startLap > 0) stops = 0; 
252 
253         while (currentLap < totalLaps) {
254             String key = currentLap + "-" + mask;
255             StintDecision decision = bestDecisions.get(key);
256             
257             if (decision == null) return null; 
258 
259             strategy.getStints().add(new Stint(decision.compound, currentLap + 1, currentLap + decision.laps));
260             
261             stops++;
262             currentLap += decision.laps;
263             mask = decision.nextMask;
264         }
265         
266         strategy.setPitStops(stops);
267         
268         // Recuperiamo il tempo totale specifico per QUESTA strategia
269         // Usiamo la chiave di partenza passata come argomento
270         String startKey = startLap + "-" + startMask;
271         Double totalTime = memo.get(startKey);
272         strategy.setTotalTime(totalTime != null ? totalTime : 0.0);
273         
274         return strategy;
275     }

277     /**
278      * Calcola il tempo totale impiegato per percorrere un certo numero di giri con una specifica gomma.
279      * <p>
280      * Simula il degrado giro per giro: al tempo base viene aggiunto il tasso di degrado
281      * accumulato per ogni giro successivo al primo.
282      * Modello lineare: Tempo(giro i) = BaseTime + (i * DegradationRate).
283      * </p>
284      *
285      * @param tyre La predizione della gomma (contiene base time e degradation rate).
286      * @param laps Il numero di giri da percorrere nello stint.
287      * @return Il tempo totale in secondi.
288      */
289     private double calculateStintTime(TyrePrediction tyre, int laps) {
290         double total = 0;
291         double currentLapTime = tyre.getBase_time();
292         for (int i = 0; i < laps; i++) {
293             total += currentLapTime;
294             currentLapTime += tyre.getDegradation_rate();
295         }
296         return total;
297     }

299     /**
300      * Classe helper interna (DTO) per memorizzare una decisione ottimale nel grafo DP.
301      * <p>
302      * Salva quale gomma è stata scelta, per quanti giri e quale sarà la prossima bitmask,
303      * permettendo al metodo {@code reconstructStrategy} di ripercorrere il cammino minimo.
304      * </p>
305      */
306     // Classe helper interna per salvare le decisioni
307     private static class StintDecision {
308         String compound;
309         int laps;
310         int nextMask;
311 
312         public StintDecision(String compound, int laps, int nextMask) {
313             this.compound = compound;
314             this.laps = laps;
315             this.nextMask = nextMask;
316         }
317     }
