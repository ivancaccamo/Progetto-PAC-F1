10  import static org.junit.jupiter.api.Assertions.*;
11  
12  class OptimizationEngineTest {
13  
14      private OptimizationEngine engine;
15      private List<TyrePrediction> mockTyres;
16  
17      @BeforeEach
18      void setUp() {
19          engine = new OptimizationEngine();
20          mockTyres = new ArrayList<>();
21  
22          // Creiamo dei dati finti (Mock) per le gomme
23          // SOFT: Veloce ma degrada presto
24          TyrePrediction soft = new TyrePrediction();
25          soft.setCompound("SOFT");
26          soft.setBase_time(90.0);       // 1m 30s
27          soft.setDegradation_rate(0.1); // perde 1 decimo a giro
28          mockTyres.add(soft);
29  
30          // MEDIUM: Media
31          TyrePrediction medium = new TyrePrediction();
32          medium.setCompound("MEDIUM");
33          medium.setBase_time(91.0);     // 1m 31s
34          medium.setDegradation_rate(0.06);
35          mockTyres.add(medium);
36  
37          // HARD: Lenta ma dura tanto
38          TyrePrediction hard = new TyrePrediction();
39          hard.setCompound("HARD");
40          hard.setBase_time(92.5);       // 1m 32.5s
41          hard.setDegradation_rate(0.02);
42          mockTyres.add(hard);
43      }
44  
45      @Test
46      void testCalculateStrategy_StandardRace() {
47          // Simuliamo una gara di 50 giri
48          int totalLaps = 50;
49  
50          List<RaceStrategy> result = engine.calculateTop3Strategies(totalLaps, mockTyres);
51  
52          // VERIFICHE (Asserts)
53          assertNotNull(result, "La lista delle strategie non deve essere null");
54          assertFalse(result.isEmpty(), "Dovrebbe trovare almeno una strategia");
55          
56          RaceStrategy best = result.get(0);
57          System.out.println("Miglior tempo trovato: " + best.getTotalTime());
58          
59          assertTrue(best.getTotalTime() > 0, "Il tempo totale deve essere positivo");
60          assertTrue(best.getPitStops() >= 1, "Per 50 giri ci aspettiamo almeno 1 sosta");
61      }
62  
63      @Test
64      void testCalculateStrategy_ShortRace() {
65          // Gara sprint di 15 giri (dovrebbe provare a farla con 0 o 1 sosta)
66          int totalLaps = 15;
67          
68          List<RaceStrategy> result = engine.calculateTop3Strategies(totalLaps, mockTyres);
69          
70          assertFalse(result.isEmpty());
71          // Controlliamo che il primo stint non superi i giri totali
72          assertTrue(result.get(0).getStints().get(0).getLaps() <= totalLaps);
73      }
74  
75      @Test
76      void testCalculateStrategy_NoTyres() {
77          // Caso limite: nessuna gomma disponibile
78          List<RaceStrategy> result = engine.calculateTop3Strategies(50, new ArrayList<>());
79          
80          assertNotNull(result);
81          assertTrue(result.isEmpty(), "Se non ci sono gomme, non ci sono strategie");
82      }
