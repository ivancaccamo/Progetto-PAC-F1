1   package com.ivancaccamo.pacf1.service;
2   
3   import com.ivancaccamo.pacf1.model.PredictionRequest;
4   import com.ivancaccamo.pacf1.model.PredictionResponse;
5   import org.springframework.stereotype.Service;
6   import org.springframework.web.client.RestTemplate;
7   import org.springframework.beans.factory.annotation.Autowired;
8   import java.util.List;
9   
10  /**
11   * Servizio responsabile della comunicazione con il microservizio esterno di Machine Learning.
12   * <p>
13   * Questa classe agisce come un client HTTP (o Adapter) che interfaccia il backend Java
14   * con l'applicazione Python (Flask). Si occupa di costruire la richiesta, inviare i dati
15   * ambientali e del circuito, e deserializzare la risposta JSON contenente le predizioni
16   * sul degrado gomme.
17   * </p>
18   *
19   * @author Team SPS-F1
20   */
21  @Service
22  public class PythonMLService {
23  
24      /**
25       * Client HTTP fornito dal framework Spring per effettuare chiamate REST.
26       */
27      @Autowired
28      private RestTemplate restTemplate;
29  
30      /**
31       * URL dell'endpoint esposto dal servizio Python Flask.
32       * Attualmente configurato per puntare all'istanza locale (localhost).
33       */
34      // Assicurati che il server Python sia acceso su questa porta!
35      private final String ML_SERVICE_URL = "http://127.0.0.1:5000/predict";
36  
37      /**
38       * Richiede al modulo Python le previsioni di degrado per il circuito e le condizioni specificate.
39       * <p>
40       * Il metodo prepara il payload JSON (incluso il set standard di mescole SOFT, MEDIUM, HARD),
41       * esegue una richiesta POST sincrona e gestisce eventuali errori di comunicazione.
42       * </p>
43       *
44       * @param circuit   Il nome del circuito su cui effettuare la predizione.
45       * @param airTemp   La temperatura dell'aria attuale.
46       * @param trackTemp La temperatura dell'asfalto attuale.
47       * @return Un oggetto {@link PredictionResponse} contenente i dati di degrado se la chiamata ha successo,
48       * {@code null} se il servizio Python non Ã¨ raggiungibile o restituisce errore.
49       */
50      public PredictionResponse getPrediction(String circuit, double airTemp, double trackTemp) {
51          List<String> compounds = List.of("SOFT", "MEDIUM", "HARD");
52          
53          PredictionRequest request = new PredictionRequest(circuit, airTemp, trackTemp, compounds);
54  
55          try {
56              return restTemplate.postForObject(ML_SERVICE_URL, request, PredictionResponse.class);
57          } catch (Exception e) {
58              System.err.println("ERRORE: Python non risponde! " + e.getMessage());
59              return null;
60          }
61      }
62  }
