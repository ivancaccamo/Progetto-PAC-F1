1   package com.ivancaccamo.pacf1.model;
2   
3   import jakarta.persistence.*;
4   import java.time.LocalDateTime;
5   import java.time.format.DateTimeFormatter;
6   
7   /**
8    * Entità JPA che rappresenta una strategia di gara salvata nel database.
9    * <p>
10   * Questa classe mappa la tabella delle strategie archiviate. Viene utilizzata
11   * per persistere i risultati delle simulazioni che l'utente decide di conservare
12   * nello storico per consultazioni future.
13   * </p>
14   *
15   * @author Team SPS-F1
16   */
17  @Entity
18  public class SavedStrategy {
19  
20      /**
21       * Identificativo univoco (Primary Key) della strategia salvata.
22       * Generato automaticamente dal database.
23       */
24      @Id
25      @GeneratedValue(strategy = GenerationType.IDENTITY)
26      private Long id;
27  
28      /**
29       * Il nome del circuito associato alla strategia (es. "Bahrain Grand Prix").
30       */
31      private String circuit;
32  
33      /**
34       * Il tempo totale stimato di gara in secondi.
35       */
36      private double totalTime;
37  
38      /**
39       * Il numero di soste ai box previste.
40       */
41      private int pitStops;
42  
43      /**
44       * Una descrizione testuale sintetica della sequenza di stint.
45       * Esempio: "Soft (18 giri) -> Medium (25 giri) -> Hard (14 giri)".
46       * Utile per la visualizzazione rapida nell'archivio senza dover ricostruire l'oggetto completo.
47       */
48      private String stintsDescription;
49  
50      /**
51       * Timestamp di creazione del record, formattato come stringa "dd-MM-yyyy HH:mm".
52       */
53      private String createdAt;
54  
55      /**
56       * Costruttore vuoto richiesto dalle specifiche JPA.
57       */
58      public SavedStrategy() {
59      }
60  
61      /**
62       * Costruttore per creare una nuova istanza di strategia da salvare.
63       *
64       * @param circuit           Il nome del circuito.
65       * @param totalTime         Il tempo totale in secondi.
66       * @param pitStops          Il numero di soste.
67       * @param stintsDescription La descrizione leggibile degli stint.
68       */
69      public SavedStrategy(String circuit, double totalTime, int pitStops, String stintsDescription) {
70          this.circuit = circuit;
71          this.totalTime = totalTime;
72          this.pitStops = pitStops;
73          this.stintsDescription = stintsDescription;
74          this.createdAt = LocalDateTime.now()
75                  .format(DateTimeFormatter.ofPattern("dd-MM-yyyy HH:mm"));
76      }
77  
78      /**
79       * Metodo di callback JPA eseguito prima della persistenza dell'entità.
80       * <p>
81       * Garantisce che il campo {@code createdAt} venga valorizzato con la data
82       * e l'ora correnti se non è stato già impostato manualmente.
83       * </p>
84       */
85      @PrePersist
86      public void onCreate() {
87          if (this.createdAt == null) {
88              this.createdAt = LocalDateTime.now()
89                      .format(DateTimeFormatter.ofPattern("dd-MM-yyyy HH:mm"));
90          }
91      }
92  
93      // ---------- GETTER & SETTER ----------
94  
95      /**
96       * Restituisce l'ID del record.
97       * @return L'identificativo univoco.
98       */
99      public Long getId() {
100         return id;
101     }
102 
103     /**
104      * Imposta l'ID del record.
105      * @param id L'identificativo univoco.
106      */
107     public void setId(Long id) {
108         this.id = id;
109     }
110 
111     /**
112      * Restituisce il nome del circuito.
113      * @return Il nome del circuito.
114      */
115     public String getCircuit() {
116         return circuit;
117     }
118 
119     /**
120      * Imposta il nome del circuito.
121      * @param circuit Il nome del circuito.
122      */
123     public void setCircuit(String circuit) {
124         this.circuit = circuit;
125     }
126 
127     /**
128      * Restituisce il tempo totale di gara salvato.
129      * @return Il tempo in secondi.
130      */
131     public double getTotalTime() {
132         return totalTime;
133     }
134 
135     /**
136      * Imposta il tempo totale di gara.
137      * @param totalTime Il tempo in secondi.
138      */
139     public void setTotalTime(double totalTime) {
140         this.totalTime = totalTime;
141     }
142 
143     /**
144      * Restituisce il numero di pit-stop salvati.
145      * @return Il numero di soste.
146      */
147     public int getPitStops() {
148         return pitStops;
149     }
150 
151     /**
152      * Imposta il numero di pit-stop.
153      * @param pitStops Il numero di soste.
154      */
155     public void setPitStops(int pitStops) {
156         this.pitStops = pitStops;
157     }
158 
159     /**
160      * Restituisce la descrizione testuale degli stint.
161      * @return La stringa descrittiva.
162      */
163     public String getStintsDescription() {
164         return stintsDescription;
165     }
166 
167     /**
168      * Imposta la descrizione testuale degli stint.
169      * @param stintsDescription La stringa descrittiva.
170      */
171     public void setStintsDescription(String stintsDescription) {
172         this.stintsDescription = stintsDescription;
173     }
174 
175     /**
176      * Restituisce la data di creazione formattata.
177      * @return La data come stringa.
178      */
179     public String getCreatedAt() {
180         return createdAt;
181     }
182 
183     /**
184      * Imposta la data di creazione.
185      * @param createdAt La data formattata come stringa.
186      */
187     public void setCreatedAt(String createdAt) {
188         this.createdAt = createdAt;
189     }
190 }
